The bool type stores boolean values - either true or false.

Syntax

var flag bool = true
let condition = false

Example

function isPositive(x int) bool
    if x > 0 'check'
        return true
    end 'check'
    return false
end 'isPositive'

function main() int
    if isPositive(5) 'test'
        return 1
    end 'test'
    return 0
end 'main'
